## / : root

`````
        
        /bin:user binaries
                |
                 /sbin:system binaries
                                     |
                                      /etc:config files
                                                      |
                                                       /dev:device files
                                                                        |
                                                                         /proc:process files
                                                                         |
                                                       /var:variable files
                                                       |
                                    /tmp:temporary files
                                    |
                    /usr:user program
                                    |
                                    /home:home directory
                                    |
                     /boot:boot files
                                     |
                                      /lib:system libraries
                                                           |
                                                            /opt:optional app
                                                                             |
                                                                              /mnt:mount directory
                                                                              |
                                                        /media:removable media
                                                        |
                                        /srv:service data
````
